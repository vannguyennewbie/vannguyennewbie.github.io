🔢 Dạng 1: Bài tập Số học C++

📘 Ví dụ: Kiểm tra số nguyên tố
#include <bits/stdc++.h>
using namespace std;
bool ktra(int x){
    if(x<2) return false;
    for(int i=2;i<=sqrt(x);i++)
        if(x%i==0) return false;
    return true;
}
int main(){
    int n; cin>>n;
    cout << (ktra(n) ? "YES" : "NO");
}

📊 Dạng 2: Bài tập Mảng

📘 Ví dụ: Đếm số chẵn trong mảng
#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,a[1000]; cin>>n;
    int dem=0;
    for(int i=0;i<n;i++){
        cin>>a[i];
        if(a[i]%2==0) dem++;
    }
    cout<<dem;
}
